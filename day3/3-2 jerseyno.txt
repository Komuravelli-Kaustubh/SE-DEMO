#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main()
{
    int N;
    cin>>N;
    vector<int> v1(N,0);
    vector<int> v2(N,0);
    vector<int> v3(N,0);
    for(int i=0;i<N;i++){
        cin>>v1[i];
    }
    
    for(int i=0;i<N;i++){
        cin>>v2[i];
    }
    
    for(int i=0;i<N;i++){
        for(int j=0;j<N;j++){
            if(v1[i]==v2[j]){
                v3[i]=j;
                v2[j]=-1;
  		break;
            }
        }
    }
    
    for(int x: v3){
        cout<<x<<" ";
    }
    
}
