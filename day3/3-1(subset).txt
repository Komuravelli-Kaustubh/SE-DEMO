from collections import Counter
n=int(input())
l=list(map(int,input().split()))
c=Counter(l)
c=dict(c)
oc=dict()
f=dict()
for v,k in enumerate(l):
    f.setdefault(k,v)
ans=50000
for i,val in enumerate(l):
    oc[val]=oc.get(val,0)+1
    if(oc[val]==max(c.values())):
        ans=min(ans, i-f[val]+1)
print(ans)
