#include <iostream>
#include <algorithm>
#include <vector>
#include <cmath>

using namespace std;

void solve(vector<int>& v,int p,int n)
{
    sort(v.begin(),v.end());
    for(int i=0;i<p;i++){
    v[n-1]=sqrt(v[n-1]);
    sort(v.begin(),v.end());
    }
    
    
    
}

int main()
{
    int n,p;
    cin>>n>>p;
    
    vector<int> v(n);
    
    for(int i=0;i<n;i++)
    {
        cin>>v[i];
    }
    
    solve(v,p,n);
    
    int sum=0;
    for(int x : v){
        sum+=x;
    }
    
    cout<<sum;
    
    
}
