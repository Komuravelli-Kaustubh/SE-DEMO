#include <iostream>
#include <unordered_map>
#include <string>

using namespace std;

int main()
{
    string s;
    cout<<"Enter string"<<endl;
    cin>>s;
    
    unordered_map<char,int> freqMap;
    int oddCount=0;
    
    
    for(char c : s)
    {
        freqMap[c]+=1;
    }
    
    for(const auto& p : freqMap)
    {
        if((p.second%2)!=0)
        {
            oddCount++;
        }
    }
    
    if(oddCount<=1)
    cout<<"true"<<endl;
    else
    cout<<"false"<<endl;
    
    
}
